using Unity.MLAgents.Actuators;
using UnityEditor;
using UnityEngine;
using RLMovie.Common;

namespace RLMovie.Editor
{
    /// <summary>
    /// Builds the starter scene for __SCENARIO_NAME__.
    /// Uses the generic backbone, then adds scenario-specific gameplay objects.
    /// </summary>
    public static class __SCENARIO_NAME__SceneBuilder
    {
        [MenuItem("RLMovie/Create __SCENARIO_NAME__ Scene")]
        public static void CreateScene()
        {
            CreateSceneSilently();

            EditorUtility.DisplayDialog(
                "__SCENARIO_NAME__ Scene",
                "Scene created.\n\nNext step:\nCustomize the gameplay logic while keeping the shared backbone intact.",
                "OK");
        }

        public static string CreateSceneSilently()
        {
            var backbone = GoldenScenarioSceneBuilder.BuildScenarioBackbone("__SCENE_PATH__");

            // --- Gameplay objects (scenario-specific) ---
            GameObject floor = GameObject.CreatePrimitive(PrimitiveType.Plane);
            floor.name = "ArenaFloor";
            floor.transform.SetParent(backbone.EnvironmentRoot.transform);
            floor.transform.localPosition = Vector3.zero;

            GameObject hero = GameObject.CreatePrimitive(PrimitiveType.Capsule);
            hero.name = "Hero";
            hero.transform.SetParent(backbone.EnvironmentRoot.transform);
            hero.transform.localPosition = new Vector3(0f, 1f, 0f);

            var rb = hero.AddComponent<Rigidbody>();
            rb.mass = 1f;
            rb.angularDamping = 0.5f;

            var agent = hero.AddComponent<RLMovie.Environments.__SCENARIO_NAME__.__AGENT_CLASS__>();
            GoldenScenarioSceneBuilder.ConfigureAgentBehavior(
                hero, "__BEHAVIOR_NAME__", 13, ActionSpec.MakeContinuous(2));

            GameObject target = GameObject.CreatePrimitive(PrimitiveType.Cube);
            target.name = "PrimaryTarget";
            target.transform.SetParent(backbone.EnvironmentRoot.transform);
            target.transform.localPosition = new Vector3(3f, 0.6f, 3f);
            target.transform.localScale = new Vector3(0.9f, 0.9f, 0.9f);

            // --- Wire agent serialized fields ---
            var agentSo = new SerializedObject(agent);
            agentSo.FindProperty("primaryTarget").objectReferenceValue = target.transform;
            agentSo.FindProperty("environmentManager").objectReferenceValue = backbone.EnvironmentManager;
            agentSo.FindProperty("startPosition").vector3Value = new Vector3(0f, 1f, 0f);
            agentSo.ApplyModifiedProperties();

            // --- Readability materials ---
            GoldenScenarioSceneBuilder.ApplyReadabilityMaterial(floor, "V2FloorReadable", new Color(0.17f, 0.21f, 0.26f));
            GoldenScenarioSceneBuilder.ApplyReadabilityMaterial(hero, "V2HeroReadable", new Color(0.92f, 0.83f, 0.28f));
            GoldenScenarioSceneBuilder.ApplyReadabilityMaterial(target, "V2TargetReadable", new Color(0.22f, 0.92f, 0.56f));

            // --- Finalize backbone with gameplay bindings ---
            backbone.FinalizeSpine(
                new[] { new ScenarioGoldenSpine.SceneRoleBinding { role = "primary_target", target = target.transform } },
                new[] { new ScenarioGoldenSpine.AgentRoleBinding { role = "hero", team = "solo", primary = true, agent = agent } },
                new[] { new ScenarioGoldenSpine.TeamRoleBinding { team = "solo", primaryAgent = agent, agents = new BaseRLAgent[] { agent } } });

            backbone.ConfigureTrainingVisualizer(agent);
            backbone.Save();

            Debug.Log("Created scene for __SCENARIO_NAME__ at __SCENE_PATH__");
            return "__SCENE_PATH__";
        }
    }
}
