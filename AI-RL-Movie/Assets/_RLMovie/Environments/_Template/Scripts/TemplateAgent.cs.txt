using RLMovie.Common;
using Unity.MLAgents.Actuators;
using Unity.MLAgents.Sensors;
using UnityEngine;

namespace RLMovie.Environments.__SCENARIO_NAME__
{
    /// <summary>
    /// V2 reference agent for __SCENARIO_NAME__.
    /// Keep the shared telemetry and heuristics, then specialize the task-specific rules later.
    /// </summary>
    public class __AGENT_CLASS__ : BaseRLAgent
    {
        [Header("=== V2 Reference Interaction ===")]
        [RequiredSharedReference("primaryTarget")]
        [SerializeField] private Transform primaryTarget;
        [RequiredSharedReference("environmentManager")]
        [SerializeField] private EnvironmentManager environmentManager;
        [SerializeField] private float moveForce = 1.0f;
        [SerializeField] private float interactionRadius = 1.35f;

        private Rigidbody _rb;
        private float _previousDistanceToTarget;

        protected override void OnAgentInitialize()
        {
            _rb = GetComponent<Rigidbody>();
        }

        protected override void OnEpisodeReset()
        {
            if (_rb != null)
            {
                _rb.linearVelocity = Vector3.zero;
                _rb.angularVelocity = Vector3.zero;
            }

            if (environmentManager != null)
            {
                transform.position = environmentManager.GetRandomPosition(1.0f);
                if (primaryTarget != null)
                {
                    primaryTarget.position = environmentManager.GetRandomEdgePosition(0.6f);
                }
            }

            _previousDistanceToTarget = GetDistanceToTarget();
        }

        protected override void CollectAgentObservations(VectorSensor sensor)
        {
            sensor.AddObservation(transform.localPosition);
            sensor.AddObservation(primaryTarget != null ? primaryTarget.localPosition : Vector3.zero);
            sensor.AddObservation(_rb != null ? _rb.linearVelocity : Vector3.zero);

            Vector3 directionToTarget = primaryTarget != null
                ? (primaryTarget.localPosition - transform.localPosition).normalized
                : Vector3.zero;
            sensor.AddObservation(directionToTarget);
            sensor.AddObservation(GetNormalizedDistance(GetDistanceToTarget()));
        }

        protected override void ExecuteActions(ActionBuffers actions)
        {
            if (_rb == null)
            {
                return;
            }

            float moveX = actions.ContinuousActions[0];
            float moveZ = actions.ContinuousActions[1];
            _rb.AddForce(new Vector3(moveX, 0f, moveZ) * moveForce, ForceMode.VelocityChange);

            float distanceToTarget = GetDistanceToTarget();
            if (distanceToTarget <= interactionRadius)
            {
                Success(1.0f, "target_reached", "Primary Target Reached");
                return;
            }

            if (environmentManager != null && environmentManager.HasFallen(transform))
            {
                Fail(-1.0f, "fell_out", "Fell Out");
                return;
            }

            float progressReward = (_previousDistanceToTarget - distanceToTarget) * 0.02f;
            AddTrackedReward(progressReward, "progress", progressReward >= 0f ? "Closing In" : "Drifting");
            AddTrackedReward(-0.0005f);

            if (_previousDistanceToTarget > interactionRadius * 1.4f && distanceToTarget <= interactionRadius * 1.15f)
            {
                RecordMarkerEvent("near_miss_setup", "Close Call Building", progressReward, AgentTelemetryEventType.Info);
            }

            _previousDistanceToTarget = distanceToTarget;
        }

        protected override void ProvideHeuristicInput(in ActionBuffers actionsOut)
        {
            var continuousActions = actionsOut.ContinuousActions;
            float moveX = 0f;
            float moveZ = 0f;

            var keyboard = UnityEngine.InputSystem.Keyboard.current;
            if (keyboard != null)
            {
                if (keyboard.upArrowKey.isPressed || keyboard.wKey.isPressed) moveZ = 1f;
                if (keyboard.downArrowKey.isPressed || keyboard.sKey.isPressed) moveZ = -1f;
                if (keyboard.rightArrowKey.isPressed || keyboard.dKey.isPressed) moveX = 1f;
                if (keyboard.leftArrowKey.isPressed || keyboard.aKey.isPressed) moveX = -1f;
            }

            continuousActions[0] = moveX;
            continuousActions[1] = moveZ;
        }

        private float GetDistanceToTarget()
        {
            return primaryTarget != null
                ? Vector3.Distance(transform.localPosition, primaryTarget.localPosition)
                : 0f;
        }

        private float GetNormalizedDistance(float distanceToTarget)
        {
            float maxDistance = environmentManager != null
                ? Mathf.Max(environmentManager.AreaRadius * 2f, 0.001f)
                : 10f;

            return Mathf.Clamp01(distanceToTarget / maxDistance);
        }
    }
}
